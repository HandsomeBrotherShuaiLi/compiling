#include<iostream>
#include<string>
#include<cmath>
#include<cstring>
using namespace std;
int main()
{
	int T,a,b,c,ans;
        long swd;
         short ded;
         //短型
	cin>>T;
       /* 词法分析器
         */
	while(T--)
	{
		cin>>a>>b>>c;//输入
		ans=0;
		if(c%gcd(a,b)!=0)
		{
			cout<<0<<endl;
		}
		else
		{
			for(int i=1;i<=(c-b)/a;i++)
			{
				if((c-a*i)/b*b==(c-a*i))
				ans++;
			}
			cout<<ans<<endl;
		}
	}
	return 0;
}
